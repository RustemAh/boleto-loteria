function render() { // Actualizar textos de cabecera document.getElementById('subtitulo').innerText = `Sorteo N° ${SORTEO_ACTIVO.n_sorteo} - ${SORTEO_ACTIVO.fecha}`; // Limpiar el monto para evitar doble $$ (Elimina el $ si ya existe en el dato) const montoLimpio = SORTEO_ACTIVO.montoMayor.toString().replace('$', ''); document.getElementById('montoM').innerText = `$${montoLimpio}`; document.getElementById('numM').innerText = SORTEO_ACTIVO.numMayor; // Mostrar aproximaciones y terminaciones en la interfaz (Opcional, para que sean visibles) console.log("Aproximaciones cargadas:", SORTEO_ACTIVO.aprox1, SORTEO_ACTIVO.aprox2); const tbody = document.getElementById('tablaBody'); const filas = SORTEO_ACTIVO.vigesimos.split(';'); let html = ""; filas.forEach(f => { if(f.includes(':')) { const [comb, monto] = f.split(':'); const [n, v] = comb.split('-'); // Limpiamos el monto de la tabla también const montoTabla = monto.toString().replace('$', ''); html += `${n}${v}$${montoTabla}`; } }); tbody.innerHTML = html; document.getElementById('resBusqueda').style.display = "none"; } function verificar() { const n = document.getElementById('inNum').value.trim(); const v = document.getElementById('inVig').value.trim(); const res = document.getElementById('resBusqueda'); if(!n || !v) return; res.style.display = "block"; const llave = `${n}-${v}`; const lista = SORTEO_ACTIVO.vigesimos.split(';'); let premio = null; // 1. Buscar en lista de vigésimos for(let f of lista) { const [c, m] = f.split(':'); if(c.trim() === llave) { premio = m; break; } } if(premio) { const pLimpio = premio.toString().replace('$', ''); res.style.background = "#d4edda"; res.style.color = "#155724"; res.innerHTML = `¡GANASTE! Vigésimo ${v} del ${n}: $${pLimpio}`; } // 2. Lógica de Aproximaciones y Terminaciones else if(n === SORTEO_ACTIVO.numMayor) { res.style.background = "#fff3cd"; res.style.color = "#856404"; res.innerHTML = "¡NUMERO GANADOR DEL PREMIO MAYOR!"; } else if (SORTEO_ACTIVO.aprox1.includes(n)) { res.style.background = "#d1ecf1"; res.style.color = "#0c5460"; res.innerHTML = "¡GANASTE 1ª APROXIMACIÓN! ($5.000.000)"; } else if (SORTEO_ACTIVO.aprox2.includes(n)) { res.style.background = "#d1ecf1"; res.style.color = "#0c5460"; res.innerHTML = "¡GANASTE 2ª APROXIMACIÓN! ($500.000)"; } else if (SORTEO_ACTIVO.terminaciones && SORTEO_ACTIVO.terminaciones.includes(n.slice(-1))) { res.style.background = "#e2e3e5"; res.style.color = "#383d41"; res.innerHTML = `PREMIO POR TERMINACIÓN "${n.slice(-1)}"`; } else { res.style.background = "#f8d7da"; res.style.color = "#721c24"; res.innerHTML = "Sin premios registrados."; } }
